064    /**
065     * Operates on CSS property signatures to come up with a simple regular
066     * expression that validates values where possible.
067     *
068     * <p>
069     * This class produces a javascript file like<pre>
070     *   var css = {
071     *       "background-image": /url("[^\"\\\(\)]+")\s+/i,
072     *       clear: /(?:none|left|right|both|inherit)\s+/i,
073     *       color: /(?:blue|red|green|fuschia)\s+/i
074     *   };
075     * </pre>
076     *
077     * <p>
078     * The {@code css} map does not contain every property in the given
079     * {@link CssSchema} since some cannot be matched efficiently by values.
080     * See comments on {@code [ a || b ]*} constructs inline below.
081     *
082     * <h3>Differences from Server Side CSS</h3>
083     * <p>
084     * Nor does it contain every option matched by the server side CSS processor.
085     * Specifically, it does not currently match<ul>
086     *   <li>function calls, e.g. {@code rect(...)} and {@code rgb(...)}
087     *   <li>the {@code unreserved-word} symbol used for font-families.
088     *       Font families need to be quoted, e.g. {@code "Helvetica"}.
089     *       The server side rewriter takes adjacent unreserved words and quotes
090     *       them, but the goal of this pattern is to allow client-side validation,
091     *       not rewriting.
092     *   <li>quoted strings where URLs are allowed.  E.g. {@code background-image}
093     *       does not match {@code "http://foo/myimage.com"}.
094     *   <li>As noted above, string literals are not allowed as URLs, but the
095     *       {@code url(...)} syntax is allowed, but only containing double-quoted
096     *       strings.  E.g. {@code url(foo.gif)} is not allowed though
097     *       {@code url("foo.gif")} is.
098     * </ul>
099     *
100     * <h3>Whitespace Between Tokens</h3>
101     * <p>
102     * The patterns in the example above all end in {@code \s+}.  This simplifies
103     * a lot of the patterns since a signature like {@code foo*} strictly translates
104     * to the regular expression {@code /(foo(\s+foo)*)?/i}.  Even if we repeated
105     * subexpressions, we would run into problems with the signature
106     * {@code a [b | c? d]? e} which could translates to a regular expression but
107     * only with non-local handling of whitespace around sub-expressions that can
108     * be empty.
109     *
110     * <p>
111     * Instead, we require every literal to be followed by one or more spaces.
112     * We can then match against CSS padded with spaces, as in<pre>
113     *   var isValid = css[cssPropertyName].test(cssText + ' ');
114     * </pre>
115     *
116     * <h3>Program Flow</h3>
117     * <p>
118     * This class examines a schema and builds a list of all allowed CSS properties.
119     * It then tries to convert each property's signature to a regular expression
120     * pattern.
121     * It may fail for some patterns, especially the aggregate ones like
122     * {@code background} which combine
123     * {@code background-image}, {@code background-style}, etc.
124     * <p>
125     * Next it optimizes the patterns it found.  This includes flattening
126     * concatenation and union operators, and moving the {@code \s+} out of unions.
127     * Optimizing {@code /((blue\s+|red\s+|green\s+)|inherit\s+)/} might yield the
128     * simpler expression {@code /(blue|red|green|inherit)\s+/}.
129     * <p>
130     * Once it has a mapping of property names to regular expressions it builds a
131     * constant pool by hashing on regular expression text.
132     * This allows properties with identical patterns such as {@code border-top} and
133     * {@code border-bottom} to share an instance.
134     * <p>
135     * Finally it builds a javascript parse tree that assigns the {@code css}
136     * namespace to an object whose keys are CSS property names, and whose
137     * values are regular expressions.
138     *
139     * <h3>Caveats</h3>
140     * <p>
141     * Some of the regular expressions do match URLs.  If valid css text contains
142     * the string 'uri' case-insensitively, then a client may need to extract and
143     * rewrite URLs.  Since all strings are double quoted, this should be doable
144     * without lexing CSS.
145     *
146     * @author mikesamuel@gmail.com
147     */
148    public class CssPropertyPatterns {
149      private final CssSchema schema;

152        this.schema = schema;

155      /**
156       * Generates a regular expression for the given signature if a simple
157       * regular expression exists.
158       * @return null if no simple regular expression exists
159       *     or the text of a Javascript regular expression like "/foo\s+/i"
160       *     that matches values of the given value with one or more trailing
161       *     whitespace characters.
162       *     If the color property only matched the literal "blue", the resulting
163       *     pattern would match "blue ".
164       */
165      public String cssPropertyToPattern(CssPropertySignature sig) {
166        Pattern p = sigToPattern(sig);
167        if (p == null) { return null; }
168        p = new Concatenation(
169            Arrays.asList(new Snippet("^\\s*"), p, new Snippet("$"))).optimize();
170        StringBuilder out = new StringBuilder();
171        out.append('/');
172        p.render(out);
173        // Since all keywords are case insensitive and we never match text inside
174        // strings.
175        out.append("/i");
176        return out.toString();
177      }
178    
179      private Pattern sigToPattern(CssPropertySignature sig) {
180        // Dispatch to a set of handlers that either append balanced content to
181        // out, or append cruft and return null.
182        if (sig instanceof CssPropertySignature.LiteralSignature) {
183          return litToPattern((CssPropertySignature.LiteralSignature) sig);
184        } else if (sig instanceof CssPropertySignature.RepeatedSignature) {
185          return repToPattern((CssPropertySignature.RepeatedSignature) sig);
186        } else if (sig instanceof CssPropertySignature.PropertyRefSignature) {
187          return refToPattern((CssPropertySignature.PropertyRefSignature) sig);
188        } else if (sig instanceof CssPropertySignature.SeriesSignature) {
189          return seriesToPattern((CssPropertySignature.SeriesSignature) sig);
190        } else if (sig instanceof CssPropertySignature.SymbolSignature) {
191          return symbolToPattern((CssPropertySignature.SymbolSignature) sig);
192        } else if (sig instanceof CssPropertySignature.SetSignature
193                   || sig instanceof CssPropertySignature.ExclusiveSetSignature) {
194          return setToPattern(sig);
195        }
196        return null;
197      }
198    
199      private static Pattern litToPattern(
200          CssPropertySignature.LiteralSignature lit) {
201        StringBuilder regex = new StringBuilder();
202        Escaping.escapeRegex(lit.getValue(), false, false, regex);
203        // Match some trailing whitespace.
204        // Since some patterns can match nothing (e.g. foo*), we make sure that
205        // all positive matches are followed by token-breaking space.
206        // The pattern as a whole can then be matched against the value with one
207        // space added at the end.
208        regex.append("\\s+");
209        return new Snippet(regex.toString());
210      }
211    
212      private Pattern repToPattern(CssPropertySignature.RepeatedSignature sig) {
213        CssPropertySignature rep = sig.getRepeatedSignature();
214        if (rep instanceof CssPropertySignature.ExclusiveSetSignature) {
215          // Can't easily handle the special semantics of [ a || b ]* allowing
216          // "a b", "b a", "a", "b", but not "a a", "b b".
217          return null;
218        }
219        Pattern repeatedPattern = sigToPattern(rep);
220        if (repeatedPattern == null) { return null; }
221        return new Repetition(repeatedPattern, sig.minCount, sig.maxCount);
222      }
223    
224      private Pattern refToPattern(CssPropertySignature.PropertyRefSignature sig) {
225        CssSchema.CssPropertyInfo p = schema.getCssProperty(sig.getPropertyName());
226        return p != null ? sigToPattern(p.sig) : null;
227      }
228    
229      private Pattern seriesToPattern(CssPropertySignature.SeriesSignature sig) {
230        List<Pattern> children = new ArrayList<Pattern>();
231        for (CssPropertySignature child : sig.children()) {
232          Pattern childP = sigToPattern(child);
233          if (childP == null) { return null; }
234          children.add(childP);
235        }
236        return new Concatenation(children);
237      }
238    
239      private Pattern symbolToPattern(CssPropertySignature.SymbolSignature sig) {
240        Name symbolName = sig.getValue();
241        Pattern builtinMatch = builtinToPattern(symbolName);
242        if (builtinMatch != null) { return builtinMatch; }
243        CssSchema.SymbolInfo s = schema.getSymbol(symbolName);
244        return s != null ? sigToPattern(s.sig) : null;
245      }
246    
247      private Pattern setToPattern(CssPropertySignature sig) {
248        if (sig.children().isEmpty()) { return null; }
249        List<Pattern> children = new ArrayList<Pattern>();
250        for (CssPropertySignature child : sig.children()) {
251          Pattern childP = sigToPattern(child);
252          if (childP != null) { children.add(childP); }
253        }
254        if (children.isEmpty()) { return null; }
255        return new Union(children);
256      }

259          = new HashMap<String, String>();
260      static {
261        // http://www.w3.org/TR/REC-CSS2/syndata.html
262        String unsignedNum = "(?:\\d+(?:\\.\\d+)?)";
263        String signedNum = "[+-]?\\d+(?:\\.\\d+)?";
264        String angleUnits = "(?:deg|g?rad)";
265        String freqUnits = "k?Hz";
266        String lengthUnits = "(?:em|ex|px|in|cm|mm|pt|pc)";
267        String timeUnits = "m?s";
268        String quotedIdentifiers = "\"\\w(?:[\\w-]*\\w)(?:\\s+\\w([\\w-]*\\w))*\"";
269        BUILTINS.put("number:0,", "0|" + unsignedNum);
270        BUILTINS.put("number:0,1", "(?:0(?:\\.[0-9]+)?|\\.[0-9]+|1(?:\\.0+)?)");
271        BUILTINS.put("number", "0|" + signedNum);
272        BUILTINS.put("percentage", "0|" + unsignedNum + "%");
273        BUILTINS.put("percentage:0,", "0|" + signedNum + "%");
274        BUILTINS.put("angle:0,", "0|" + unsignedNum + angleUnits);
275        BUILTINS.put("angle", "0|" + signedNum + angleUnits);
276        BUILTINS.put("frequency", "0|" + unsignedNum + freqUnits);
277        BUILTINS.put("length:0,", "0|" + unsignedNum + lengthUnits);
278        BUILTINS.put("length", "0|" + signedNum + lengthUnits);
279        BUILTINS.put("time:0,", "0|" + unsignedNum + timeUnits);
280        BUILTINS.put("time", "0|" + signedNum + timeUnits);
281        BUILTINS.put("integer", "-?\\d+");
282        BUILTINS.put("integer:0,", "\\d+");

286        BUILTINS.put("uri", "url\\(\"[^\\(\\)\\\\\\\"\\r\\n]+\"\\)");

288      private Pattern builtinToPattern(Name name) {
289        String pattern = BUILTINS.get(name.getCanonicalForm());
290        if (pattern == null && name.getCanonicalForm().contains(":")) {
291          System.err.println("Failing detail check on " + name);
292        }
293        return pattern != null ? new Snippet(pattern + "\\s+") : null;
294      }

296      private static interface Pattern {
297        /** Returns a simpler but equivalent Pattern. */
298        Pattern optimize();
299        /** Appends javascript regular expression to the given buffer. */
300        void render(StringBuilder out);
301        /** A Pattern suffix that matches all strings matched by this pattern. */
302        String tail();
303        /**
304         * A pattern that matches the same content as this pattern but without
305         * matching the last n characters.
306         */
307        Pattern subtractTail(int n);
308      }

310      private static final class Snippet implements Pattern {
311        final String patternSnippet;
312        Snippet(String patternSnippet) {
313          this.patternSnippet = patternSnippet;
314        }
315        public Pattern optimize() {
316          return this;
317        }
318        public void render(StringBuilder out) { out.append(patternSnippet); }
319        public String tail() { return patternSnippet; }
320        public Pattern subtractTail(int n) {
321          return new Snippet(
322              patternSnippet.substring(0, patternSnippet.length() - n));
323        }
324        @Override
325        public boolean equals(Object o) {
326          if (!(o instanceof Snippet)) { return false; }
327          return patternSnippet.equals(((Snippet) o).patternSnippet);
328        }
329        @Override
330        public int hashCode() {
331          return patternSnippet.hashCode() ^ 0x4482de40;
332        }
333      }

335      private static final class Concatenation implements Pattern {
336        final List<Pattern> children;
337        Concatenation(List<Pattern> children) { this.children = children; }
338        public Pattern optimize() {
339          List<Pattern> newChildren = new ArrayList<Pattern>();
340          for (Pattern child : children) {
341            child = child.optimize();
342            if (child instanceof Concatenation) {
343              Concatenation childCat = (Concatenation) child;
344              newChildren.addAll(childCat.children);
345            } else if (!(child instanceof Snippet
346                         && "".equals(((Snippet) child).patternSnippet))) {
347              newChildren.add(child);
348            }
349          }
350          if (newChildren.size() == 1) { return newChildren.get(0); }
351          return new Concatenation(newChildren);
352        }
353        public void render(StringBuilder out) {
354          for (Pattern n : children) { n.render(out); }
355        }
356        public String tail() {
357          return children.get(children.size() - 1).tail();
358        }
359        public Pattern subtractTail(int n) {
360          List<Pattern> newChildren = new ArrayList<Pattern>();
361          int last = children.size() - 1;
362          newChildren.addAll(children.subList(0, last));
363          newChildren.add(children.get(last).subtractTail(n));
364          return new Concatenation(newChildren);
365        }
366        @Override
367        public boolean equals(Object o) {
368          if (!(o instanceof Concatenation)) { return false; }
369          return children.equals(((Concatenation) o).children);
370        }
371        @Override
372        public int hashCode() {
373          return children.hashCode() ^ 0xd18b17e;
374        }
375      }

377      private static class Union implements Pattern {
378        final List<Pattern> children;
379        Union(List<Pattern> children) { this.children = children; }
380        public Pattern optimize() {
381          List<Pattern> newChildren = new ArrayList<Pattern>();
382          for (Pattern child : children) {
383            child = child.optimize();
384            if (child instanceof Union) {
385              Union childUnion = (Union) child;
386              newChildren.addAll(childUnion.children);
387            } else {
388              newChildren.add(child);
389            }
390          }
391          int n = newChildren.size();
392          if (n == 1) { return newChildren.get(0); }
393          if (n != 0) {  // a\\s+|b\\s+|c\\s+  -> (a|b|c)\\s+
394            // Tail optimize since most patterns will end with \\s+.
395            Pattern child0 = newChildren.get(0);
396            String tail = child0.tail();
397            for (Pattern child : newChildren.subList(1, n)) {
398              if ("".equals(tail)) { break; }
399              tail = commonSuffix(tail, child.tail());
400            }
401            if (!"".equals(tail)) {
402              for (int i = 0; i < n; ++i) {
403                newChildren.set(i, newChildren.get(i).subtractTail(tail.length()));
404              }
405              Pattern opt = new Concatenation(
406                  Arrays.asList(new Union(newChildren), new Snippet(tail)))
407                  .optimize();
408              return opt;
409            }
410          }
411          // Remove duplicates
412          Set<Pattern> seen = new HashSet<Pattern>();
413          for (Iterator<Pattern> it = newChildren.iterator(); it.hasNext();) {
414            if (!seen.add(it.next())) { it.remove(); }
415          }
416          if (n == 1) { return newChildren.get(0); }
417          return new Union(newChildren);
418        }
419        public void render(StringBuilder out) {
420          out.append("(?:");
421          for (int i = 0, n = children.size(); i < n; ++i) {
422            if (i != 0) { out.append('|'); }
423            children.get(i).render(out);
424          }
425          out.append(')');
426        }
427        public String tail() { return ""; }
428        public Pattern subtractTail(int n) {
429          throw new UnsupportedOperationException();
430        }
431        @Override
432        public boolean equals(Object o) {
433          if (!(o instanceof Union)) { return false; }
434          return children.equals(((Union) o).children);
435        }
436        @Override
437        public int hashCode() {
438          return children.hashCode() ^ 0x11b79bed;
439        }
440      }

442      private static final class Repetition implements Pattern {
443        final Pattern repeated;
444        final int min, max;
445        Repetition(Pattern repeated, int min, int max) {
446          this.repeated = repeated;
447          this.min = min;
448          this.max = max;
449        }
450        public Pattern optimize() {
451          return new Repetition(repeated.optimize(), min, max);
452        }
453        public void render(StringBuilder out) {
454          out.append("(?:");
455          repeated.render(out);
456          out.append(')');
457          if (max == Integer.MAX_VALUE) {
458            switch (min) {
459              case 0: out.append('*'); return;
460              case 1: out.append('+'); return;
461            }
462          } else if (max == 1 && min == 0) {
463            out.append('?');
464            return;
465          }
466          out.append('{').append(min).append(',').append(max).append('}');
467        }
468        public String tail() { return ""; }
469        public Pattern subtractTail(int n) {
470          throw new UnsupportedOperationException();
471        }
472        @Override
473        public boolean equals(Object o) {
474          if (!(o instanceof Repetition)) { return false; }
475          Repetition that = (Repetition) o;
476          return this.min == that.min && this.max == that.max
477              && this.repeated.equals(that.repeated);
478        }
479        @Override
480        public int hashCode() {
481          return (repeated.hashCode() + 31 * (min + 31 * max)) ^ 0xa3c916;
482        }
483      }

485      public static String commonSuffix(String a, String b) {
486        int m = a.length(), n = b.length();
487        int k = Math.min(m, n);
488        int i = 0;
489        while (i < k && a.charAt(m - i - 1) == b.charAt(n - i - 1)) { ++i; }
490        return a.substring(m - i, m);
491      }

493      public static void generatePatterns(CssSchema schema, Appendable out)
494          throws IOException {
495        FilePosition unk = FilePosition.UNKNOWN;
496        CssPropertyPatterns pp = new CssPropertyPatterns(schema);
497        List<CssSchema.CssPropertyInfo> props
498            = new ArrayList<CssSchema.CssPropertyInfo>(schema.getCssProperties());
499        Collections.sort(
500            props, new Comparator<CssSchema.CssPropertyInfo>() {
501              public int compare(CssSchema.CssPropertyInfo a,
502                                 CssSchema.CssPropertyInfo b) {
503                return a.name.compareTo(b.name);
504              }
505            });
506        Map<String, int[]> constantPoolMap = new HashMap<String, int[]>();
507        List<Pair<CssSchema.CssPropertyInfo, String>> patterns
508            = new ArrayList<Pair<CssSchema.CssPropertyInfo, String>>();
509        List<RegexpLiteral> constantPool = new ArrayList<RegexpLiteral>();
510    
511        for (CssSchema.CssPropertyInfo prop : props) {
512          String pattern = pp.cssPropertyToPattern(prop.sig);
513          if (!schema.isPropertyAllowed(prop.name)) { continue; }
514          if (pattern != null && !"(?:inherit\\s+)".equals(pattern)) {
515            patterns.add(Pair.pair(prop, pattern));
516            // Keep track of which patterns appear more than once so we can use
517            // a constant pool.
518            int[] pool = constantPoolMap.get(pattern);
519            if (pool == null) {
520              constantPoolMap.put(pattern, new int[] { -1 });
521            } else if (pool[0] == -1) {
522              pool[0] = constantPool.size();
523              constantPool.add(new RegexpLiteral(unk, pattern));
524            }
525          }
526        }
527    
528        Declaration constantPoolDecl = null;
529        if (!constantPool.isEmpty()) {
530          constantPoolDecl = (Declaration) QuasiBuilder.substV(
531              "var c = @constantPool;",
532              "constantPool", new ArrayConstructor(unk, constantPool));
533        }
534        List<Pair<Literal, Expression>> members
535            = new ArrayList<Pair<Literal, Expression>>();
536        List<Pair<Literal, Expression>> alternates
537            = new ArrayList<Pair<Literal, Expression>>();
538        for (Pair<CssSchema.CssPropertyInfo, String> p : patterns) {
539          int poolIndex = constantPoolMap.get(p.b)[0];
540          Expression re = poolIndex < 0
541              ? new RegexpLiteral(unk, p.b)
542              : (Expression) QuasiBuilder.substV(
543                  "c[@i]", "i", new IntegerLiteral(unk, poolIndex));
544          Literal name = StringLiteral.valueOf(unk, p.a.name.getCanonicalForm());
545          members.add(Pair.pair(name, re));
546    
547          String dom2property = propertyNameToDom2Property(p.a.name);
548          ArrayConstructor altNames = null;
549          for (String altDom2Property : p.a.dom2properties) {
550            if (altDom2Property.equals(dom2property)) { continue; }
551            if (altNames == null) {
552              altNames = new ArrayConstructor(
553                  unk, Collections.<Expression>emptyList());
554              alternates.add(Pair.pair(
555                  (Literal) StringLiteral.valueOf(unk, dom2property),
556                  (Expression) altNames));
557            }
558            altNames.appendChild(StringLiteral.valueOf(unk, altDom2Property));
559          }
560        }
561        ObjectConstructor cssPropConstructor = new ObjectConstructor(unk, members);
562        ObjectConstructor alternateNames = new ObjectConstructor(unk, alternates);
563    
564        ParseTreeNode js = QuasiBuilder.substV(
565            "var css = { properties: (function () {"
566            + "  @constantPoolDecl?;"
567            + "  return @cssPropConstructor;"
568            + "})(), alternates: @alternates };",
569            "constantPoolDecl", constantPoolDecl,
570            "cssPropConstructor", cssPropConstructor,
571            "alternates", alternateNames);
572        TokenConsumer tc = js.makeRenderer(out, null);
573        js.render(new RenderContext(new MessageContext(), tc));
574        tc.consume(";");
575        tc.noMoreTokens();
576        out.append("\n");
577      }

579      public static class Builder implements BuildCommand {
580        public void build(List<File> inputs, List<File> deps, File output)
581            throws IOException {
582          File symbolsAndPropertiesFile = null;
583          File functionsFile = null;
584          for (File input : inputs) {
585            if (input.getName().endsWith(".json")) {
586              if (symbolsAndPropertiesFile == null) {
587                symbolsAndPropertiesFile = input;
588              } else if (functionsFile == null) {
589                functionsFile = input;
590              } else {
591                throw new IOException("Unused input " + input);
592              }
593            }
594          }
595          if (symbolsAndPropertiesFile == null) {
596            throw new IOException("No JSON whitelist for CSS Symbols + Properties");
597          }
598          if (functionsFile == null) {
599            throw new IOException("No JSON whitelist for CSS Functions");
600          }
601    
602          FilePosition sps = FilePosition.startOfFile(new InputSource(
603              symbolsAndPropertiesFile.getAbsoluteFile().toURI()));
604          FilePosition fns = FilePosition.startOfFile(new InputSource(
605              functionsFile.getAbsoluteFile().toURI()));
606    
607          MessageContext mc = new MessageContext();
608          mc.addInputSource(sps.source());
609          mc.addInputSource(fns.source());
610          MessageQueue mq = new EchoingMessageQueue(
611              new PrintWriter(new OutputStreamWriter(System.err), true), mc, false);
612    
613          Set<File> inputsAndDeps = new HashSet<File>();
614          for (File f : inputs) { inputsAndDeps.add(f.getAbsoluteFile()); }
615          for (File f : deps) { inputsAndDeps.add(f.getAbsoluteFile()); }
616    
617          ImportResolver resolver = new AllowedFileResolver(inputsAndDeps);
618    
619          CssSchema schema;
620          try {
621            schema = new CssSchema(
622                ConfigUtil.loadWhiteListFromJson(
623                    sps.source().getUri(), resolver, mq),
624                ConfigUtil.loadWhiteListFromJson(
625                    fns.source().getUri(), resolver, mq));
626          } catch (ParseException ex) {
627            ex.toMessageQueue(mq);
628            throw (IOException) new IOException("Failed to parse schema")
629                .initCause(ex);
630          }
631    
632          Writer out = new OutputStreamWriter(
633              new FileOutputStream(output), "UTF-8");
634          String currentDate = "" + new Date();
635          if (currentDate.indexOf("*/") >= 0) { throw new RuntimeException(); }
636          out.write("/* Copyright Google Inc.\n");
637          out.write(" * Licensed under the Apache Licence Version 2.0\n");
638          out.write(" * Autogenerated at " + currentDate + "\n");
639          out.write(" */\n");
640          try {
641            generatePatterns(schema, out);
642          } finally {
643            out.close();
644          }
645        }
646      }

648      /**
649       * Converts a css property name to a javascript identifier, e.g.
650       * {@code background-color} => {@code backgroundColor}.
651       */
652      static String propertyNameToDom2Property(Name cssPropertyName) {
653        String lcaseDashed = cssPropertyName.getCanonicalForm();
654        int dash = lcaseDashed.indexOf('-');
655        if (dash < 0) { return lcaseDashed; }
656        StringBuilder sb = new StringBuilder(lcaseDashed.length());
657        int written = 0;
658        do {
659          sb.append(lcaseDashed, written, dash);
660          written = dash + 1;
661          if (written < lcaseDashed.length()) {
662            sb.append(Strings.toUpperCase(
663                lcaseDashed.substring(written, written + 1)));
664            ++written;
665          }
666          dash = lcaseDashed.indexOf('-', written);
667        } while (dash >= 0);
668        sb.append(lcaseDashed, written, lcaseDashed.length());
669        return sb.toString();
670      }
